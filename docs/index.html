Quick Start

HTML 
<div id="carousel-container">
  <div class="slide">Slide 1</div>
  <div class="slide">Slide 2</div>
  <div class="slide">Slide 3</div>
</div>
JavaScript 
const carousel = horizontalLoop("#carousel-container", {
  responsive: {
    0: { items: 1 },
    768: { items: 2 },
    1024: { items: 3 }
  },
  speed: 1.2,
  gap: "20px",
  draggable: true,
  autoplayDelay: 3
});
