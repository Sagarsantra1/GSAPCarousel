const logoCarousel = horizontalLoop("#logo-carousel", {
          speed: 2,
          gap: "20px",
          paused: false,
          repeat: -1,
        });

// Pause on hover
document.querySelector("#logo-carousel").addEventListener("mouseenter", () => {
  logoCarousel.pause();
});
document.querySelector("#logo-carousel").addEventListener("mouseleave", () => {
  logoCarousel.resume();
});
